    사용방법
    1. 다운로드 : jQuery 홈페이지에서 다운로드
        제이쿼리 다운로드하기
        
        

    2. CDN(주소참조활용) : Content Delivery Network (인터넷에서 컨텐츠를 가지고 옴)
        
        

    [기본문법]
    1. id 문법
        $(document).ready(function(){//문서에서 어떤 행동을 할 준비를 함
            $("#아이디이름").click(function(){//id로 아이디이름을 가진 태그를 클릭했을 때 실행할 동작
                동작에 관련된 코드 작성;
            })
        })

    2. 이벤트리스너 문법(onclick, keyon, onchange, keydown, ...)
        function 이벤트리스너에작성한이름() {
            $("#아이디명").기능작성();
        }

    3. 다른 아이디값에서 원하는 태그를 선택할 때 사용하는 문법
        $("#아이디명 태그명")
    
    [선택자문법]
    1. css()
        $("태그명 또는 #아이디명 또는 .클래스명").css("배경", 색상);
    
    2.toggle()
        $("태그명 또는 #아이디명 또는 .클래스명").toggle("속도조절");//속도생략가능
    
    3. slideToggle()
        토글과 비슷하나 주로 이미지를 보이고 숨길 때 사용
    
    4. hide() / show()
        보이기  숨기기
    
    5. append()
        태그나 값을 추가할 때 사용
    
    6. val()
        input, textarea, select 같은 폼에서 태그의 값을 가지고 오거나 설정하는데 사용

        값을 가져올 때
            $(태그명|#아이디명|.클래스명).val()
                $("input").val()
                $("#아이디명").val()
                $(".클래스명").val()

        값을 설정할 때
            $(태그명|#아이디명|.클래스명).val("새로운 값 설정")
                $("input").val("새로운 값 설정")
                $("#아이디명").val("새로운 값 설정")
                $(".클래스명").val("새로운 값 설정")

    7. this
            현재 실행중인 함수(태그)를 나타냄
            this 이벤트가 발생한 태그를 참조
            $("#인풋필드").on("input", function() {
                console.log(this); //이벤트가 발생한 input태그를 바라봄
            })

    8. on()
        이벤트 기능, 하나 이상의 이벤트를 처리
        $(태그명|#아이디명|.클래스명).on("event", function() {

        })
        event에 속하는 유형 : click, input, mouseenter, mouseover, ...

        on 같은 경우에는 on에 해당하는 이벤트가 많기 때문에 on을 작성한 후
        on("이벤트명", function() {}) 처럼 이벤트 명을 작성해줘야 하지만
        다른 속성들은 "이벤트명" 작성하지 않아도 됨

    9. keyup()
        사용자가 입력 창(input, textarea, ...)에 키를 입력할 때 마다 발생하는 이벤트를 감지
        키보드의 키가 눌릴 때 마다 발생
    
    10. .each()
        for-each 문 처럼 지정한 태그나 클래스명이나 아이디명을 반복하는 기능
        $("li").each(function() {

        })
        li태그에 해당하는 모든 값을 가져와서 반복해서 수행하고자 하는 기능을 진행
    
    11. indexOf()
        문자열 또는 문자에서 첫 번째 발생한 인덱스를 반환
        찾지 못할 경우 에러를 최소화하기 위해서 -1을 반환

        //index           0       1
        const 과일들 = ["사과", "바나나"];
        const 바나나찾기 = 과일들.indexOf("바나나");
        console.log(바나나찾기); //출력 1
        const 수박찾기 = 과일들.indexOf("수박");
        console.log(수박찾기); //출력 -1
            //java였다면 수박이 과일들에 존재하지 않으므로
            //에러가 출력되어야 하지만
            //javascript는 에러를 최소화하기 위해 -1을 출력
        
        indexOf를 했을 때 검색을 지우고 나면 문자열 ""(빈칸)

        keyup 이벤트에서 검색어가 빈 문자열이면 모든 항목을 보이도록 설정되어 있음
        indexOf "" 모든 값에 잠재적으로 저장되어 있음
